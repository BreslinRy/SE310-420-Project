CREATE TABLE tally(
	votes_cast VARCHAR(100)
	voter_id VARCHAR(100),
	has_voted CHAR();

CREATE TABLE candidates(
	candidate_id VARCHAR(100),
	candidate_party VARCHAR(100),
	candidate_name VARCHAR(100),
	candidate_descrip LONGTEXT)
	candidate_tally VARCHAR(100);

CREATE TABLE voter(
	voter_id VARCHAR(100),
	has_voted CHAR(),
	voter_selection_id VARCHAR(100);
